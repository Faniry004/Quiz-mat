<script>
    const quizData = [
        { q: "Quelle est la limite de \( e^x \) en \( -\infty \) ?", options: ["1", "0", "+\infty", "-\infty"], correct: 1 },
        { q: "Dérivée de \( 3x^2 \) ?", options: ["3x", "6x", "6", "x^2"], correct: 1 }
    ];

    const cards = [
        { front: "Formule suite arithmétique", back: "\( u_n = u_0 + nr \)" },
        { front: "Dérivée de \( \ln(x) \)", back: "\( 1/x \)" }
    ];

    function showPage(pageId, btn) {
        document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
        document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));
        document.getElementById('page-' + pageId).classList.add('active');
        btn.classList.add('active');
        if(window.MathJax) MathJax.typesetPromise();
    }

    let cardIdx = 0; let isFlipped = false;
    function updateCard() {
        const f = document.getElementById('flashcard');
        f.innerHTML = isFlipped ? cards[cardIdx].back : cards[cardIdx].front;
        if(window.MathJax) MathJax.typesetPromise();
    }
    function flipCard() { isFlipped = !isFlipped; updateCard(); }
    function nextCard() { cardIdx = (cardIdx + 1) % cards.length; isFlipped = false; updateCard(); }
    function prevCard() { cardIdx = (cardIdx - 1 + cards.length) % cards.length; isFlipped = false; updateCard(); }

    function initQuiz() {
        const container = document.getElementById('quiz-content');
        container.innerHTML = `
            <div class="top-bar">
                <div class="progress-dots"><div class="dot active"></div><div class="dot"></div></div>
                <span style="opacity: 0.6">1/10</span>
            </div>
            <h2 style="font-weight: 300; margin-bottom: 30px;">${quizData[0].q}</h2>
            ${quizData[0].options.map((o, i) => `<button class="btn-opt" onclick="alert('Réponse enregistrée')"><span style="opacity:0.3; margin-right:10px;">${i+1}</span> ${o}</button>`).join('')}
        `;
        if(window.MathJax) MathJax.typesetPromise();
    }

    window.onload = () => { initQuiz(); updateCard(); };
</script>
